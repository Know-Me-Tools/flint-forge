## What this change delivers

A new `flint_a2ui` PostgreSQL schema installed as a SQL migration (not a pgrx extension). This schema is the Flint-invented global A2UI component registry — a value-add layer on top of the official A2UI protocol (https://a2ui.org/specification/v0.9.1-a2ui/). The official A2UI spec defines only a "Basic Catalog" (Text, Button, Row); this registry extends the catalog model with PostgreSQL storage, JSONB extensibility, pgvector semantic search, and per-application scoping.

### Schema tables

```sql
CREATE SCHEMA IF NOT EXISTS flint_a2ui;

-- Primary component catalog
CREATE TABLE flint_a2ui.components (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    slug            text NOT NULL UNIQUE,
    category        text NOT NULL CHECK (category IN (
                        'layout', 'data-display', 'input', 'action', 'navigation', 'feedback', 'system'
                    )),
    primitive_type  text NOT NULL,    -- A2UI catalog type string (e.g. 'DataGrid', 'TextInput')
    schema          jsonb NOT NULL,   -- JSON Schema for prop validation
    is_base         boolean NOT NULL DEFAULT false,
    application_id  uuid,             -- NULL = available to all applications
    description     text,
    usage_examples  jsonb,
    design_tokens   jsonb,
    metadata        jsonb NOT NULL DEFAULT '{}',
    created_at      timestamptz NOT NULL DEFAULT now(),
    updated_at      timestamptz NOT NULL DEFAULT now()
);

-- Application registry
CREATE TABLE flint_a2ui.applications (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    slug            text NOT NULL UNIQUE,
    name            text NOT NULL,
    description     text,
    owner_id        uuid,
    jwt_claims_template jsonb NOT NULL DEFAULT '{}',
    catalog_id      text UNIQUE,      -- URI for A2UI catalog reference
    metadata        jsonb NOT NULL DEFAULT '{}',
    created_at      timestamptz NOT NULL DEFAULT now(),
    updated_at      timestamptz NOT NULL DEFAULT now()
);

-- Seed base applications
INSERT INTO flint_a2ui.applications (slug, name, description, is_system)
VALUES
    ('flint-admin',      'Flint Admin',      'Administrative interface', true),
    ('flint-playground', 'Flint Playground', 'Developer sandbox',        true);

-- Design system records (Flint Design Bundle, internal format)
CREATE TABLE flint_a2ui.design_systems (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    application_id  uuid REFERENCES flint_a2ui.applications(id) ON DELETE CASCADE,
    slug            text NOT NULL UNIQUE,
    name            text NOT NULL,
    bundle_url      text,             -- URL to design bundle index (Flint Design Bundle format)
    tokens          jsonb NOT NULL DEFAULT '{}',
    metadata        jsonb NOT NULL DEFAULT '{}',
    created_at      timestamptz NOT NULL DEFAULT now()
);

-- pgvector embeddings for semantic search
CREATE TABLE flint_a2ui.embeddings (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    component_id    uuid NOT NULL REFERENCES flint_a2ui.components(id) ON DELETE CASCADE,
    embedding       vector(1536) NOT NULL,
    entity_type     text NOT NULL DEFAULT 'component',
    aspect          text NOT NULL DEFAULT 'description',  -- 'description'|'usage'|'props'
    model           text NOT NULL DEFAULT 'text-embedding-3-large',
    created_at      timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX idx_embeddings_hnsw ON flint_a2ui.embeddings
    USING hnsw (embedding vector_cosine_ops)
    WITH (m = 16, ef_construction = 64);

-- Prop schemas per component (separate for query efficiency)
CREATE TABLE flint_a2ui.schemas (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    component_id    uuid NOT NULL REFERENCES flint_a2ui.components(id) ON DELETE CASCADE,
    schema_version  text NOT NULL DEFAULT '1.0',
    json_schema     jsonb NOT NULL,
    created_at      timestamptz NOT NULL DEFAULT now()
);

-- DB table → component bindings (auto-generated by trigger)
CREATE TABLE flint_a2ui.bindings (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    table_schema    text NOT NULL,
    table_name      text NOT NULL,
    component_id    uuid NOT NULL REFERENCES flint_a2ui.components(id),
    binding_type    text NOT NULL CHECK (binding_type IN ('form', 'grid', 'detail', 'card')),
    auto_generated  boolean NOT NULL DEFAULT true,
    config          jsonb NOT NULL DEFAULT '{}',
    created_at      timestamptz NOT NULL DEFAULT now(),
    UNIQUE (table_schema, table_name, binding_type)
);

-- Append-only event log
CREATE TABLE flint_a2ui.events (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    application_id  uuid REFERENCES flint_a2ui.applications(id),
    event_type      text NOT NULL,
    actor           text,
    object          text,
    action          text,
    result          boolean,
    details         jsonb,
    created_at      timestamptz NOT NULL DEFAULT now()
);

-- Event-driven assembly rules
CREATE TABLE flint_a2ui.assembly_rules (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    application_id  uuid NOT NULL REFERENCES flint_a2ui.applications(id),
    event_type      text NOT NULL,
    event_filter    jsonb NOT NULL DEFAULT '{}',
    assembly_config jsonb NOT NULL,
    priority        int NOT NULL DEFAULT 100,
    is_active       boolean NOT NULL DEFAULT true,
    created_at      timestamptz NOT NULL DEFAULT now(),
    updated_at      timestamptz NOT NULL DEFAULT now()
);

-- Role hierarchy
CREATE TABLE flint_a2ui.roles (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    application_id  uuid NOT NULL REFERENCES flint_a2ui.applications(id),
    slug            text NOT NULL,
    name            text NOT NULL,
    parent_role_id  uuid REFERENCES flint_a2ui.roles(id),
    keto_namespace  text,
    keto_relation   text,
    created_at      timestamptz NOT NULL DEFAULT now(),
    UNIQUE (application_id, slug)
);

-- User → role assignments
CREATE TABLE flint_a2ui.role_assignments (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    application_id  uuid NOT NULL REFERENCES flint_a2ui.applications(id),
    role_id         uuid NOT NULL REFERENCES flint_a2ui.roles(id),
    user_id         text NOT NULL,   -- from JWT claims
    granted_by      text,
    granted_at      timestamptz NOT NULL DEFAULT now(),
    UNIQUE (application_id, role_id, user_id)
);
```

### RLS policies

```sql
-- Components: base components visible to all; app-specific components scoped by role
ALTER TABLE flint_a2ui.components ENABLE ROW LEVEL SECURITY;

CREATE POLICY component_access ON flint_a2ui.components
    FOR ALL USING (
        is_base = true
        OR application_id IS NULL
        OR application_id IN (
            SELECT application_id FROM flint_a2ui.role_assignments
            WHERE user_id = (current_setting('app.jwt_claims', true)::jsonb)->'flint'->>'user_id'
        )
    );

-- Events: append-only (no UPDATE/DELETE)
ALTER TABLE flint_a2ui.events ENABLE ROW LEVEL SECURITY;
CREATE POLICY event_insert ON flint_a2ui.events FOR INSERT WITH CHECK (true);
CREATE POLICY event_select ON flint_a2ui.events FOR SELECT USING (
    application_id IS NULL
    OR application_id IN (
        SELECT application_id FROM flint_a2ui.role_assignments
        WHERE user_id = (current_setting('app.jwt_claims', true)::jsonb)->'flint'->>'user_id'
    )
);
-- Explicitly deny UPDATE/DELETE
CREATE POLICY event_no_update ON flint_a2ui.events FOR UPDATE USING (false);
CREATE POLICY event_no_delete ON flint_a2ui.events FOR DELETE USING (false);
```

### Semantic search function (stub, populated by p5-c004)

```sql
CREATE OR REPLACE FUNCTION flint_a2ui.semantic_search(
    query_embedding vector(1536),
    result_limit    int DEFAULT 10
) RETURNS TABLE (
    component_id    uuid,
    slug            text,
    similarity      float
) LANGUAGE sql STABLE AS $$
    SELECT
        c.id,
        c.slug,
        1 - (e.embedding <=> query_embedding) AS similarity
    FROM flint_a2ui.embeddings e
    JOIN flint_a2ui.components c ON c.id = e.component_id
    WHERE e.aspect = 'description'
    ORDER BY e.embedding <=> query_embedding
    LIMIT result_limit;
$$;
```

## Security notes

- `app.jwt_claims` GUC (set by p3-c008 extended GUC propagation) is the source of truth for RLS user_id
- `current_setting('app.jwt_claims', true)` — the `true` flag means NULL is returned (not error) if GUC is unset; guards against service-role queries that don't set the GUC
- Events table is append-only at the RLS level; no application code can DELETE or UPDATE event records
